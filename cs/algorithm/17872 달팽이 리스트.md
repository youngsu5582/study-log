# 달팽이 리스트

- 문제링크 : https://www.acmicpc.net/problem/17827
```java
// 21:53 - 22:20
public static void main(final String[] args) throws IOException {
    // 선형 단방향 연결리스트 - 노드 번호 연결된 순서대로 1,2, ... N

    // N번 노드는 아무도 가르키지 않음

    // N 번 노드가 1번 제외 노드 가리켜 사이클 이룰 시, 달팽이 리스트
    // K 번 이동하면 어떤 값이 있는가??

    // 10 10 3
    // N M V
    // V: N 번 노드가 가르키는 노드 번호 ( 2 <= V <= N - 1번부터 시작 )

    // 13 2 5 11 7 8 2 4 9 10 -> 5 -> 11 -> 7 -> 8 -> 2 -> 4 -> 9 -> 10 -> 5 -> ...
    // 13부터 니까, 1번 이동은 2
    // 2번 이동은 5
    // 9 번 이동은 10
    // 11번 이동은 11

    // 즉, 적절히 subString 의 느낌

    // 적절히 나누기로 처리 안되는가...?

    // 7번 이동하면 동일
    // 10 - 3 = 7
    // 5 -> 11 -> 7 -> 8 -> 2 -> 4 -> 9 -> 10 -> 5

    // N - 1 이동하면 끝에 도달
    // N 번 이동하면 사이클에 도달

    // 10 - 3 + 1 = 8

    // 49?
    // 10번째 : 5,
    // 11번째 :
    // 18번째 : 5
    // 26번째 : 5
    // 49번째 : 10

    BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    int[] nmc = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();
    int n = nmc[0];
    int m = nmc[1];
    int c = nmc[2];
    StringTokenizer st = new StringTokenizer(br.readLine());

    int[] ary = new int[n];
    for (int i = 0; i < n; i++) {
        ary[i] = Integer.parseInt(st.nextToken());
    }

    // 3번재, zero index 이므로 -1
    int cycleStart = c - 1;
    for (int i = 0; i < m; i++) {
        int question = Integer.parseInt(br.readLine());
        // 배열보다 더 작으면 가볍게 처리
        if (question < n) {
            System.out.println(ary[question]);
            continue;
        }
        // 10 ->
        // 10 -10 = 0 % ( 10 - 3 ) = 0
        // 0 + 2 = 2

        // 11 -> 11
        // 11 - 10 = 1
        // 1 % 7 + 2 = 3
        // 11

        // 19
        // 19 - 10 = 9
        // 9 % 8 = 1
        // 1 + 2 = 3

        // 25
        // 25 - 10 = 15
        // 15 % 8 = 7
        // 7 + 2 = 9
        // 10
        System.out.println(ary[(question - n) % (n - cycleStart) + cycleStart]);
    }
}
```
