import java.io.*;
import java.util.*;
import java.util.stream.*;

// 23:55 -

// k 는 13
// 어마어마하게 늘어난다...?

// 7 -> 7
// 8 -> 28
static BufferedWriter bw;

public static void main(final String[] args) throws IOException {

    BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    bw = new BufferedWriter(new OutputStreamWriter(System.out));
    String line = br.readLine();
    while (!line.equals("0")) {
        StringTokenizer st = new StringTokenizer(line);
        int n = Integer.parseInt(st.nextToken());

        int[] ary = new int[n];
        for (int i = 0; i < n; i++) {
            ary[i] = Integer.parseInt(st.nextToken());
        }

        // 7 - 6 = 1
        for (int i = 0; i <= n - 6; i++) {
            var temp = new ArrayList<Integer>();
            temp.add(ary[i]);
            permutation(ary, temp, i + 1);
        }

        // 전부 종료되면 구분자로 \n
        bw.write("\n");

        // 적절히 다시 초기화
        set = new ArrayList<>();
        line = br.readLine();
    }
    bw.flush();
    bw.close();
}

static List<List<Integer>> set = new ArrayList<>();

private static void permutation(int[] ary, List<Integer> result, int index) throws IOException {
    // 끝.
    if (result.size() == 6) {
        bw.write(parse(result));
        bw.write("\n");
        return;
    }
    for (int i = index; i < ary.length; i++) {
        result.add(ary[i]);
        permutation(ary, new ArrayList<>(result), i + 1);
        result.remove(result.size() - 1);
    }
}

private static String parse(List<Integer> list) {
    StringBuilder sb = new StringBuilder();
    for (Integer element : list) {
        sb.append(element).append(" ");
    }
    return sb.toString();
}
