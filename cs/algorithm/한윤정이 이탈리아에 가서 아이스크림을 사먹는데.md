# 한윤정이 이탈리아에 가서 아이스크림을 사먹는데

- 문제 링크 : https://www.acmicpc.net/problem/2422
어떻게 해야 더 잘 풀었을라나..

```java
import java.io.*;
import java.util.*;
import java.util.stream.*;

public class Main {


    public static void main(final String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        // 최대 3가지 선택
        // 섞어먹으면 안되는 조합은 (즉, 2개) 여러 개 전달

        // 결국, 두개 선택하고 나머지 한개 선택하는 형식으로
        int[] nm = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).toArray();
        int n = nm[0];
        int m = nm[1];

        Set<List<Integer>> set = new HashSet<>();

        for (int i = 0; i < m; i++) {
            var temp = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt).boxed()
                    .sorted()
                    .collect(Collectors.toList());
            set.add(temp);
        }

        int answer = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                List<Integer> temp = new ArrayList<>();
                temp.add(i);
                temp.add(j);

                if (set.contains(temp)) {
                    continue;
                }
                for (int k = j + 1; k <= n; k++) {
                    temp.add(k);
                    if (check(temp, set)) {
                        answer++;
                    }
                    temp.remove(temp.size() - 1);
                }
            }
        }
        System.out.println(answer);
    }

    private static boolean check(List<Integer> list, Set<List<Integer>> set) {
        if (list.size() != 3) {
            return false;
        }
        int x = list.get(0);
        int y = list.get(1);
        int z = list.get(2);
        return !set.contains(List.of(x, y)) && !set.contains(List.of(y, z)) && !set.contains(List.of(x, z));

    }
}
```
