# 회의실 배정

- 문제링크 : https://www.acmicpc.net/problem/1931

```java
import java.io.*;
import java.util.*;

public class Main {

    // 12:10

    // N개의 회의에 대해 회의실 사용표
    // 시작시간과 끝 시간 주어짐

    // 회의가 겹치지 않게 사용할 수 있는 회의 최대 개수

    // 한번 시작하면, 중간에 중단 X
    // 회의가 끝나는 동시에 다음 회의 시작
    // 시작 시간과 끝나는 시간 같을수도 있음 - 시작하자마자 끝남

    private static class Tuple implements Comparable<Tuple> {

        int left;
        int right;

        Tuple(int left, int right) {
            this.left = left;
            this.right = right;
        }

        // 두번째가 낮은 순
        // 첫번째도 낮은 순
        public int compareTo(Tuple o) {
            if (this.right == o.right) {
                // 왼쪽 > 오른쪽
                // 양수
                return Integer.compare(this.left, o.left);
            }
            return Integer.compare(this.right, o.right);
        }

        public String toString() {
            return "Left: " + this.left + "\t" + "Right: " + this.right;
        }


    }

    public static void main(final String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(br.readLine());
        //
        List<Tuple> tuples = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            int[] temp = Arrays.stream(br.readLine().split(" ")).mapToInt(Integer::parseInt)
                .toArray();
            tuples.add(new Tuple(temp[0], temp[1]));
        }
        Collections.sort(tuples);

        // 오른쪽을 기준으로 이동
        int leftIndex = 0;
        int result = 0;
        for (Tuple t : tuples) {
            // 어차피 못 참가하는 방
            if (leftIndex > t.left) {
                continue;
            }
            leftIndex = t.right;
            result++;
        }
        System.out.println(result);

        // 가장 많은 회의를 하려면?
        // -> 결국 가장 빨리 끝내는걸 기준으로 다 처리해야한다
        // 어차피, 그 순서가 아니라면 최적의 해를 찾지 못함

        // 더 늦게 끝나는걸 처리하는데 어떻게 앞에걸 처리??

        // 1 4
        // 3 5
        // 0 6
        // 5 7
        // 3 8
        // 5 9
        // 6 10
        // 8 11
        // 8 12
        // 2 13
        // 12 14

        // 1 4
        // 5 7
        // 8 11
        // 12 14
        // => 4

        // 1 4
        // 1 5
        // 2 5
        // 2 6
        // 4 9
        // 5 6
        // 6 7
        // 6 8

        // ->
        // 1 4
        // 1 5
        // 2 5
        // 2 6
        // 5 6
        // 6 7
        // 6 8

        // 1 4
        // 5 6
        // 6 7
        // => 3

    }
}
```
